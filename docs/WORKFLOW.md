## 🔄 重试机制

```javascript
// chat.js - 请求重试
async function sendMessageWithRetry(query, maxRetries = 3) {
  for (let i = 0; i < maxRetries; i++) {
    try {
      const response = await fetch('/api/chat/stream', {
        method: 'POST',
        body: JSON.stringify({ query, selected_sheets }),
        signal: AbortSignal.timeout(60000) // 60秒超时
      });
      
      if (!response.ok) {
        throw new Error(`HTTP ${response.status}`);
      }
      
      return response;
    } catch (error) {
      if (i === maxRetries - 1) throw error;
      console.warn(`重试 ${i + 1}/${maxRetries}...`);
      await sleep(1000 * (i + 1)); // 指数退避
    }
  }
}
```
